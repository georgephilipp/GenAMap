017    package org.apache.commons.math.analysis.interpolation;
018    
019    import org.apache.commons.math.MathException;
020    import org.apache.commons.math.analysis.TrivariateRealFunction;
021    
022    /**
023     * Interface representing a trivariate real interpolating function where the
024     * sample points must be specified on a regular grid.
025     *
026     * @version $Revision$ $Date$
027     * @since 2.2
028     */
029    public interface TrivariateRealGridInterpolator {
030        /**
031         * Computes an interpolating function for the data set.
032         *
033         * @param xval All the x-coordinates of the interpolation points, sorted
034         * in increasing order.
035         * @param yval All the y-coordinates of the interpolation points, sorted
036         * in increasing order.
037         * @param zval All the z-coordinates of the interpolation points, sorted
038         * in increasing order.
039         * @param fval the values of the interpolation points on all the grid knots:
040         * {@code fval[i][j][k] = f(xval[i], yval[j], zval[k])}.
041         * @return a function that interpolates the data set.
042         * @throws org.apache.commons.math.exception.NoDataException if any of the arrays has zero length.
043         * @throws org.apache.commons.math.exception.DimensionMismatchException if the array lengths are inconsistent.
044         * @throws MathException if arguments violate assumptions made by the
045         *         interpolation algorithm.
046         */
047        TrivariateRealFunction interpolate(double[] xval, double[] yval, double[] zval, double[][][] fval)
048            throws MathException;
049    }
