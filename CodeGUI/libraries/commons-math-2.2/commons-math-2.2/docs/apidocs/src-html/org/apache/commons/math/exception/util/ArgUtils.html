017    package org.apache.commons.math.exception.util;
018    
019    import java.util.List;
020    import java.util.ArrayList;
021    
022    /**
023     * Utility class for transforming the list of arguments passed to
024     * constructors of exceptions.
025     *
026     * @version $Revision$ $Date$
027     * @since 2.2
028     */
029    public class ArgUtils {
030        /**
031         * Private constructor
032         */
033        private ArgUtils() {
034        }
035    
036        /**
037         * Transform a multidimensional array into a one-dimensional list.
038         *
039         * @param array Array (possibly multidimensional).
040         * @return a list of all the {@code Object} instances contained in
041         * {@code array}.
042         */
043        public static Object[] flatten(Object[] array) {
044            final List<Object> list = new ArrayList<Object>();
045            if (array != null) {
046                for (Object o : array) {
047                    if (o instanceof Object[]) {
048                        for (Object oR : flatten((Object[]) o)) {
049                            list.add(oR);
050                        }
051                    } else {
052                        list.add(o);
053                    }
054                }
055            }
056            return list.toArray();
057        }
058    }
