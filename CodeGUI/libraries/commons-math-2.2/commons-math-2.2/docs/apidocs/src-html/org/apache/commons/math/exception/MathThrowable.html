017    package org.apache.commons.math.exception;
018    
019    import java.util.Locale;
020    
021    import org.apache.commons.math.exception.util.Localizable;
022    
023    /**
024    * Interface for commons-math throwables.
025    *
026    * @version $Revision: 1035475 $ $Date: 2010-11-15 23:39:25 +0100 (lun. 15 nov. 2010) $
027    * @since 2.2
028    */
029    public interface MathThrowable {
030    
031        /** Gets the localizable pattern used to build the specific part of the message of this throwable.
032         * @return localizable pattern used to build the specific part of the message of this throwable
033         */
034        Localizable getSpecificPattern();
035    
036        /** Gets the localizable pattern used to build the general part of the message of this throwable.
037         * @return localizable pattern used to build the general part of the message of this throwable
038         */
039        Localizable getGeneralPattern();
040    
041        /** Gets the arguments used to build the message of this throwable.
042         * @return the arguments used to build the message of this throwable
043         */
044        Object[] getArguments();
045    
046        /** Gets the message in a specified locale.
047         * @param locale Locale in which the message should be translated
048         * @return localized message
049         */
050        String getMessage(final Locale locale);
051    
052        /** Gets the message in a conventional US locale.
053         * @return localized message
054         */
055        String getMessage();
056    
057        /** Gets the message in the system default locale.
058         * @return localized message
059         */
060        String getLocalizedMessage();
061    
062    }
