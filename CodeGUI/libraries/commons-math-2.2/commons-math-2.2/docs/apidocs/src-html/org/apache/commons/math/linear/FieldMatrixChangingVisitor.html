018    package org.apache.commons.math.linear;
019    
020    import org.apache.commons.math.FieldElement;
021    import org.apache.commons.math.linear.MatrixVisitorException;
022    
023    /**
024     * Interface defining a visitor for matrix entries.
025     *
026     * @param <T> the type of the field elements
027     * @version $Revision: 1073158 $ $Date: 2011-02-21 22:46:52 +0100 (lun. 21 f??vr. 2011) $
028     * @since 2.0
029     */
030    public interface FieldMatrixChangingVisitor<T extends FieldElement<?>> {
031    
032        /**
033         * Start visiting a matrix.
034         * <p>This method is called once before any entry of the matrix is visited.</p>
035         * @param rows number of rows of the matrix
036         * @param columns number of columns of the matrix
037         * @param startRow Initial row index
038         * @param endRow Final row index (inclusive)
039         * @param startColumn Initial column index
040         * @param endColumn Final column index (inclusive)
041         */
042        void start(int rows, int columns,
043                   int startRow, int endRow, int startColumn, int endColumn);
044    
045        /**
046         * Visit one matrix entry.
047         * @param row row index of the entry
048         * @param column column index of the entry
049         * @param value current value of the entry
050         * @return the new value to be set for the entry
051         * @throws MatrixVisitorException if something wrong occurs
052         */
053        T visit(int row, int column, T value)
054            throws MatrixVisitorException;
055    
056        /**
057         * End visiting a matrix.
058         * <p>This method is called once after all entries of the matrix have been visited.</p>
059         * @return the value that the <code>walkInXxxOrder</code> must return
060         */
061        T end();
062    
063    }
