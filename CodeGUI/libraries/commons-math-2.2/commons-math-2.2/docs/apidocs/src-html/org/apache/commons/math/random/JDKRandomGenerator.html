017    package org.apache.commons.math.random;
018    
019    import java.util.Random;
020    
021    /**
022     * Extension of <code>java.util.Random</code> to implement
023     * {@link RandomGenerator}.
024     *
025     * @since 1.1
026     * @version $Revision: 811685 $ $Date: 2009-09-05 19:36:48 +0200 (sam. 05 sept. 2009) $
027     */
028    public class JDKRandomGenerator extends Random implements RandomGenerator {
029    
030        /** Serializable version identifier. */
031        private static final long serialVersionUID = -7745277476784028798L;
032    
033        /** {@inheritDoc} */
034        public void setSeed(int seed) {
035            setSeed((long) seed);
036        }
037    
038        /** {@inheritDoc} */
039        public void setSeed(int[] seed) {
040            // the following number is the largest prime that fits in 32 bits (it is 2^32 - 5)
041            final long prime = 4294967291l;
042    
043            long combined = 0l;
044            for (int s : seed) {
045                combined = combined * prime + s;
046            }
047            setSeed(combined);
048        }
049    
050    }
