018    package org.apache.commons.math.ode.jacobians;
019    
020    import org.apache.commons.math.ode.DerivativeException;
021    import org.apache.commons.math.ode.FirstOrderDifferentialEquations;
022    
023    
024    /** This interface represents {@link ParameterizedODE
025     * first order differential equations} with parameters and partial derivatives.
026     *
027     * @see FirstOrderIntegratorWithJacobians
028     *
029     * @version $Revision: 1073158 $ $Date: 2011-02-21 22:46:52 +0100 (lun. 21 f??vr. 2011) $
030     * @since 2.1
031     * @deprecated as of 2.2 the complete package is deprecated, it will be replaced
032     * in 3.0 by a completely rewritten implementation
033     */
034    @Deprecated
035    public interface ODEWithJacobians extends FirstOrderDifferentialEquations {
036    
037        /** Get the number of parameters.
038         * @return number of parameters
039         */
040        int getParametersDimension();
041    
042        /** Compute the partial derivatives of ODE with respect to state.
043         * @param t current value of the independent <I>time</I> variable
044         * @param y array containing the current value of the state vector
045         * @param yDot array containing the current value of the time derivative of the state vector
046         * @param dFdY placeholder array where to put the jacobian of the ODE with respect to the state vector
047         * @param dFdP placeholder array where to put the jacobian of the ODE with respect to the parameters
048         * @throws DerivativeException this exception is propagated to the caller if the
049         * underlying user function triggers one
050         */
051        void computeJacobians(double t, double[] y, double[] yDot, double[][] dFdY, double[][] dFdP)
052            throws DerivativeException;
053    
054    }
